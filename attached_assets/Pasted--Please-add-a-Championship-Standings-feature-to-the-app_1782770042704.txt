"Please add a Championship Standings feature to the app. This feature stores driver and constructor standings data that the user manually enters and updates themselves. It never calls any API or AI. Here is every detail:
How it works

The user manually enters driver names and their points after each race. The app sorts them automatically and displays a clean standings table. There is a Refresh button that re-sorts the data the user has already entered. There is an Add Driver button to add new entries. There is an Edit button on each row to update points after a race. All data is saved to localStorage so it persists between sessions.
The data structure

Store standings in localStorage like this:
javascriptconst STANDINGS_KEY = 'f1_driver_standings'
const CONSTRUCTOR_KEY = 'f1_constructor_standings'
const STANDINGS_META_KEY = 'f1_standings_meta'

function getDriverStandings() {
  try {
    return JSON.parse(localStorage.getItem(STANDINGS_KEY)) || getDefaultDriverStandings()
  } catch {
    return getDefaultDriverStandings()
  }
}

function getConstructorStandings() {
  try {
    return JSON.parse(localStorage.getItem(CONSTRUCTOR_KEY)) || getDefaultConstructorStandings()
  } catch {
    return getDefaultConstructorStandings()
  }
}

function getStandingsMeta() {
  try {
    return JSON.parse(localStorage.getItem(STANDINGS_META_KEY)) || {
      lastUpdated: null,
      lastRace: null,
      round: null,
      season: new Date().getFullYear()
    }
  } catch {
    return { lastUpdated: null, lastRace: null, round: null, season: new Date().getFullYear() }
  }
}

function saveDriverStandings(standings) {
  localStorage.setItem(STANDINGS_KEY, JSON.stringify(standings))
}

function saveConstructorStandings(standings) {
  localStorage.setItem(CONSTRUCTOR_KEY, JSON.stringify(standings))
}

function saveStandingsMeta(meta) {
  localStorage.setItem(STANDINGS_META_KEY, JSON.stringify(meta))
}

The sort function — this is what the Refresh button triggers
javascriptfunction sortStandings(standings) {
  return standings
    .sort((a, b) => {
      // Sort by points descending
      if (b.points !== a.points) return b.points - a.points
      // Tiebreak by wins descending
      return b.wins - a.wins
    })
    .map((entry, index) => ({ ...entry, pos: index + 1 }))
}

function refreshStandings() {
  // Sort drivers
  const drivers = getDriverStandings()
  const sortedDrivers = sortStandings(drivers)
  saveDriverStandings(sortedDrivers)
  
  // Sort constructors
  const constructors = getConstructorStandings()
  const sortedConstructors = sortStandings(constructors)
  saveConstructorStandings(sortedConstructors)
  
  // Update last refreshed time
  renderStandings()
  
  // Show a brief confirmation
  showToast('Standings refreshed!')
}

function saveEdit(index, type) {
  const points = parseInt(document.getElementById('edit-points').value) || 0
  const wins = parseInt(document.getElementById('edit-wins').value) || 0
  
  if (type === 'drivers') {
    const standings = getDriverStandings()
    standings[index].points = points
    standings[index].wins = wins
    saveDriverStandings(sortStandings(standings))
  } else {
    const standings = getConstructorStandings()
    standings[index].points = points
    standings[index].wins = wins
    saveConstructorStandings(sortStandings(standings))
  }
  
  closeEditModal()
  renderStandings()
  showToast('Updated and sorted!')
}

function closeEditModal() {
  const modal = document.getElementById('edit-modal')
  if (modal) modal.remove()
}

let addType = 'drivers'

function setAddType(type) {
  addType = type
  document.getElementById('add-type-drivers').style.background = type === 'drivers' ? '#e10600' : 'transparent'
  document.getElementById('add-type-drivers').style.color = type === 'drivers' ? 'white' : '#555'
  document.getElementById('add-type-constructors').style.background = type === 'constructors' ? '#e10600' : 'transparent'
  document.getElementById('add-type-constructors').style.color = type === 'constructors' ? 'white' : '#555'
}

function saveNewEntry() {
  const points = parseInt(document.getElementById('add-points').value) || 0
  const wins = parseInt(document.getElementById('add-wins').value) || 0
  
  if (addType === 'drivers') {
    const name = document.getElementById('add-name').value.trim()
    const team = document.getElementById('add-team').value.trim()
    const nationality = document.getElementById('add-nationality').value.trim()
    if (!name) return
    
    const standings = getDriverStandings()
    standings.push({ pos: standings.length + 1, driver: name, team, nationality: nationality || '🏁', points, wins })
    saveDriverStandings(sortStandings(standings))
  } else {
    const team = document.getElementById('add-name').value.trim()
    if (!team) return
    
    const standings = getConstructorStandings()
    standings.push({ pos: standings.length + 1, team, points, wins, color: '#333' })
    saveConstructorStandings(sortStandings(standings))
  }
  
  closeAddModal()
  renderStandings()
  showToast('Added and sorted!')
}

function closeAddModal() {
  const modal = document.getElementById('add-modal')
  if (modal) modal.remove()
}
