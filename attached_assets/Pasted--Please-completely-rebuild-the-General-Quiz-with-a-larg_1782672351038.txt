The rendering logic for each question type
Build a renderQuestion function that checks the question type and renders it differently:
javascriptfunction renderQuestion(q, questionNumber, totalQuestions, onAnswer) {
  if (q.type === 'standard' || q.type === 'radio' || 
      q.type === 'helmet' || q.type === 'moment' || 
      q.type === 'livery' || q.type === 'sponsor') {
    return renderStandardQuestion(q, questionNumber, totalQuestions, onAnswer)
  } else if (q.type === 'career') {
    return renderCareerQuestion(q, questionNumber, totalQuestions, onAnswer)
  } else if (q.type === 'graph') {
    return renderGraphQuestion(q, questionNumber, totalQuestions, onAnswer)
  }
}
For career path questions render the teams as a horizontal timeline with arrows between each team name and the years underneath:
javascriptfunction renderCareerQuestion(q, ...) {
  const timeline = q.career.map((team, i) => `
    <div style="display:flex;align-items:center;gap:4px">
      <div style="background:#111;color:white;padding:6px 10px;border-radius:6px;font-size:12px;font-weight:500;text-align:center">
        <div>${team}</div>
        ${q.years[i] ? `<div style="font-size:9px;color:#e10600;margin-top:2px">${q.years[i]}</div>` : ''}
      </div>
      ${i < q.career.length - 1 ? '<div style="color:#e10600;font-size:16px">→</div>' : ''}
    </div>
  `).join('')
  
  return `
    <div style="overflow-x:auto;padding-bottom:8px;margin-bottom:16px">
      <div style="display:flex;align-items:center;gap:4px;min-width:max-content">
        ${timeline}
      </div>
    </div>
    ${renderOptions(q, onAnswer)}
  `
}
For finishing position graph questions render a simple visual bar chart showing the championship positions across years. Lower position number means higher on the chart. P1 is at the top, P10 is at the bottom:
javascriptfunction renderGraphQuestion(q, ...) {
  const maxPos = Math.max(...q.positions.map(p => p.pos))
  const chartHeight = 120
  
  const bars = q.positions.map(p => {
    const barHeight = ((maxPos - p.pos + 1) / maxPos) * chartHeight
    const color = p.pos === 1 ? '#e10600' : p.pos <= 3 ? '#e65100' : '#1565c0'
    return `
      <div style="display:flex;flex-direction:column;align-items:center;gap:4px">
        <div style="font-size:10px;font-weight:700;color:${color}">P${p.pos}</div>
        <div style="width:28px;height:${barHeight}px;background:${color};border-radius:3px 3px 0 0;min-height:8px"></div>
        <div style="font-size:9px;color:var(--color-text-tertiary)">${p.year}</div>
      </div>
    `
  }).join('')
  
  return `
    <div style="margin-bottom:8px;font-size:11px;color:var(--color-text-secondary)">Championship finishing positions by season</div>
    <div style="display:flex;align-items:flex-end;gap:6px;margin-bottom:16px;padding:12px;background:var(--color-background-secondary);border-radius:8px;overflow-x:auto">
      ${bars}
    </div>
    ${renderOptions(q, onAnswer)}
  `
}
