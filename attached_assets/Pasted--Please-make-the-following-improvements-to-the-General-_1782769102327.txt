Change 1 — Weekly quiz that resets on Monday
The general quiz should only be playable once per week. It resets every Monday at midnight. Use localStorage to track whether the player has completed this week's quiz:
javascriptfunction hasCompletedWeeklyQuiz() {
  const lastCompleted = localStorage.getItem('generalQuizCompleted')
  if (!lastCompleted) return false
  
  const lastDate = new Date(lastCompleted)
  const now = new Date()
  
  // Find the most recent Monday
  const dayOfWeek = now.getDay()
  const daysSinceMonday = dayOfWeek === 0 ? 6 : dayOfWeek - 1
  const lastMonday = new Date(now)
  lastMonday.setDate(now.getDate() - daysSinceMonday)
  lastMonday.setHours(0, 0, 0, 0)
  
  return lastDate >= lastMonday
}

function markWeeklyQuizCompleted() {
  localStorage.setItem('generalQuizCompleted', new Date().toISOString())
}

function getNextMondayCountdown() {
  const now = new Date()
  const nextMonday = new Date(now)
  const daysUntilMonday = now.getDay() === 1 ? 7 : (8 - now.getDay()) % 7 || 7
  nextMonday.setDate(now.getDate() + daysUntilMonday)
  nextMonday.setHours(0, 0, 0, 0)
  
  const diff = nextMonday - now
  const days = Math.floor(diff / (1000 * 60 * 60 * 24))
  const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60))
  const mins = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60))
  
  return `${days}d ${hours}h ${mins}m`
}

function highlightMatch(text, query) {
  const lower = text.toLowerCase()
  const queryLower = query.toLowerCase()
  const index = lower.indexOf(queryLower)
  if (index === -1) return text
  // Use underline instead of red bold text
  return text.substring(0, index) +
    '<span style="text-decoration:underline;text-underline-offset:2px;">' + 
    text.substring(index, index + query.length) + 
    '</span>' +
    text.substring(index + query.length)
}
Change 7 — Weekly quiz completion flow
When the player finishes the general quiz call markWeeklyQuizCompleted and show the locked screen with their score. Also update the streak counter:
javascriptfunction completeWeeklyQuiz(finalScore) {
  const previousStreak = getWeeklyStreak()
  
  // Check if they completed last week to maintain streak
  const lastCompleted = localStorage.getItem('generalQuizCompleted')
  const streakValid = lastCompleted ? checkStreakValid(lastCompleted) : false
  
  if (streakValid) {
    localStorage.setItem('generalQuizStreak', previousStreak + 1)
  } else {
    localStorage.setItem('generalQuizStreak', 1)
  }
  
  localStorage.setItem('generalQuizLastScore', finalScore)
  markWeeklyQuizCompleted()
  
  renderWeeklyQuizLocked(finalScore)
}

function checkStreakValid(lastCompletedISO) {
  const lastDate = new Date(lastCompletedISO)
  const now = new Date()
  const diffDays = (now - lastDate) / (1000 * 60 * 60 * 24)
  return diffDays <= 14 // Allow up to 2 weeks gap before streak breaks
}
